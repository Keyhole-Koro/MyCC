int min(int a, int b) {
    if (a < b) {
        return a;
    } else if (a == b) {
        return 0;
    }
    return b;
}

int fact(int n) {
    if (n <= 1) {
        return 1;
    }
    return n * fact(n - 1);
}

int main() {
    int x = 3;
    int y = 5;
    int z;
    z = min(x, y);

    int f = fact(z);

    if (f > 5) {
        while (z > 0) {
            z = z - 1;
        }
        return f;
    } else if (f == 5) {
        for (int i = 0; i < 5; i = i + 1) {
            x = x + i;
        }
        return x;
    } else {
        z = z + 10;
    }

    return 0;
}
